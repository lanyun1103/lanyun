小结：

我使用思维导图把所有的功能写出来，并且每个功能通过测试可以使用（测试附图，肯定是可以正确运行使用的），在主函数中使用try进行错误处理。

在查找一个int和str不能在枚举类中进行比较的一个bug中，以为append进列表中的全部都是int，但是后来一步一步debug知道进去的全是str，并且可以使用sort进行比较。这是一个出现在我修改了一个学生的成绩后，无法进行排序的bug（不修改是可以的），于是将修改的成绩去掉强制转换成int也就是修改为默认input的类型str。从而问题解决。

                            '巨蟹座', '狮子座', '处女座',

            (1, 20), (2, 19), (3, 21), (4, 20), (5, 21),
            (6, 22), (7, 23), (8, 23), (9, 23), (10, 24),
            (11, 23), (12, 22)

    def get_zodiac_name(self, month, day):
        i = 0
        for data in self.zodiac_day:

            if month == 12 and day > 22:
                return self.zodiac_name[0]
            if month == data[0] and day <= data[1]:
                return self.zodiac_name[i]
            elif month == data[0] and day > data[1]:
                return self.zodiac_name[i + 1]
            i = i + 1

    def get_sx(self, year):
        return self.shengxiao[(year - 4) % 12]

        if (user.birth_year, user.birth_month, user.birth_day) == (

# 迭代器
class WeatherInterator(Iterator):
    def __init__(self, cities):
        self.cities = cities
        self.index = 0

    def getWeather(self, city):
        r = requests.get('http://wthrcdn.etouch.cn/weather_mini?city=' + city)
        data = r.json()['data']['forecast'][0]
        return '{0}:{1},{2},{3},{4}'.format(city, data['low'], data['high'], data['type'], data['fengxiang'])

    def __next__(self):
        if self.index == len(self.cities):
            raise StopIteration
        city = self.cities[self.index]
        self.index += 1
        return self.getWeather(city)

# 可迭代对象
class WeatherIterable(Iterable):
    def __init__(self, cities):
        self.cities = cities

    def __iter__(self):
        return WeatherInterator(self.cities)

class Requester():
    def __init__(self, isbn):
        self.url = "http://search.dangdang.com?key="
        self.isbn = isbn
        self.agent = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/83.0.4103.116 Safari/537.36'
        }

    def start(self):
        response = requests.post(self.url + self.isbn, headers=self.agent)
        content = response.text
        soup = BeautifulSoup(content, 'html.parser')
        # 初始化一个最小值
        names = []
        address = []
        price = []
        for i in range(50):
            for tag in soup.find_all('li', class_='line' + str(i)):
                names.append(str(tag.find('a').get('title')))
                address.append(str(tag.find('a').get('href')))
                price.append(float(str(tag.find('span', class_='search_now_price').get_text()).strip('¥')))

        return names,address,price

class MainConrole():
    def __init__(self):
        self.students = []
        self.teacher = []
        self.admin = []

        self.filer = FileControler()

    def init(self):
        self.students = self.filer.stu_read_out(self.students)
        self.teacher = self.filer.tea_read_out(self.teacher)
        self.admin = self.filer.admin_read_out(self.admin)

        self.menu_view.show_main_menu()

    def ender(self):

        self.menu_view.show_end()

    def change_stu_password(self, username, tar_psw, m):
        for stu in m.students:
            if username == stu.username:
                stu.password = tar_psw
                # 修改入文件
                self.filer.write_in_all(m)

    # 修改教师密码
    def change_tea_password(self, username, tar_psw, m):
        for tea in m.teacher:
            if username == tea.username:
                tea.password = tar_psw
                # print('修改成功')
                # 修改入文件
                self.filer.write_in_all(m)

    def change_score(self, name, score, m):
        for stu in m.students:
            if stu.name == name:
                stu.score = score
                self.filer.write_in_all(m)

    #成绩排序
    def order_score(self,m):
        score_list = []
        for stu in m.students:
            score_list.append(stu.score)

        score_list.sort()
        score_list.reverse()
        self.menu_view.show_ordered_score(score_list)

            return False

    def dx_geter(username,password,userlist):
        for info in userlist:
            if username == info.username:
                return info
